    --header "Accept: application/json"

);

const headers = {

}).then(response => response.json());

    "http://nayan.pro/api/categories" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \

    \"name\": \"voluptatem\",
    \"des\": \"ea\"
}"

    "http://nayan.pro/api/categories"
);

    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "name": "voluptatem",
    "des": "ea"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
